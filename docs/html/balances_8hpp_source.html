49void add_balance(eosio::name source, eosio::name username, eosio::asset quantity,
50 eosio::name contract) {
51 // Если баланс не найден, создаем новую запись.
52 // В противном случае, увеличиваем существующий баланс.
53
54
55 eosio::check(username != ""_n, "В поле memo должен быть указан получатель баланса");
56 balances_index balances(source, username.value);
57
58 auto balances_by_contract_and_symbol =
59 balances.template get_index<"byconsym"_n>();
60 auto contract_and_symbol_index =
61 combine_ids(contract.value, quantity.symbol.code().raw());
62

64 balances_by_contract_and_symbol.find(contract_and_symbol_index);

69 b.contract = contract;
70 b.quantity = quantity;
71 });
72 } else {
73 balances_by_contract_and_symbol.modify(

84 balances_index balances(source, username.value);

86 auto balances_by_contract_and_symbol =
87 balances.template get_index<"byconsym"_n>();
88 auto contract_and_symbol_index =
89 combine_ids(contract.value, quantity.symbol.code().raw());
90
91 auto balance =
92 balances_by_contract_and_symbol.find(contract_and_symbol_index);
93
94 eosio::check(balance != balances_by_contract_and_symbol.end(),
95 "Баланс не найден");
96
97 eosio::check(balance->quantity >= quantity, "Недостаточный баланс");
98
99 if (balance->quantity == quantity) {
100
101 balances_by_contract_and_symbol.erase(balance);
102
103 } else {
104
105 balances_by_contract_and_symbol.modify(
106 balance, source, [&](auto &b) { b.quantity -= quantity; });
107 }
